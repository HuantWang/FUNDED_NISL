static int
tapdisk_daemon_run(void)
{
	int nfds, max;
	fd_set readfds;

	while (1) {
		max = tapdisk_daemon_set_fds(&readfds);

		nfds = pselect(max + 1, &readfds, NULL, NULL, NULL,
			       &tapdisk_daemon.sigunmask);
		if (nfds < 0) {
			if (errno != EINTR)
				PERROR("select");
		}

		if (nfds > 0)
			tapdisk_daemon_check_fds(&readfds);

		tapdisk_daemon_reap_channels();
	}

	return 0;
}
