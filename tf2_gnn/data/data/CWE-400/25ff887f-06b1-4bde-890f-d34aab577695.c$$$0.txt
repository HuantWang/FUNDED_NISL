int
chgproccnt(uid_t uid, int diff)
{
	struct uidinfo *uip;

	if (diff == 0)
		return 0;

	if ((uip = getuidinfo(uid)) != NULL) {
		uip->ui_proccnt += diff;
		KASSERT(uip->ui_proccnt >= 0);
		if (uip->ui_proccnt > 0)
			return uip->ui_proccnt;
		else {
			if (uip->ui_sbsize == 0)
				freeuidinfo(uip);
			return 0;
		}
	} else {
		if (diff < 0)
			panic("chgproccnt: lost user %lu", (unsigned long)uid);
		uip = allocuidinfo(uid);
		uip->ui_proccnt = diff;
		return uip->ui_proccnt;
	}
}
