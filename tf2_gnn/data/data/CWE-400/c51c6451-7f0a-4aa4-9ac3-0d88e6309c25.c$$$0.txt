void client_close(int fd)
{
	struct client *c = client_find(fd);
	if (c) {
		vdebug("client %d // %s session ending (sent %d bytes)", fd, c->addr, c->bytes);
		vdebug("closing connection from %s, fd %d", c->addr, fd);
		c->fd = -1;
		close(fd);
	}
}
